BUFF的结构

//BUFF编号，不用管
		int32 ID = 1;
	//开发者用的名字，不用管
		FName dName;
	//玩家用的名字
		FName pName;
	//玩家看的描述
		FString miaoshu;
	//玩家看的图标
		UTexture2D* tubiao;
	//BUFF持续时间
		float LifeTime = 0;
	//BUFF是否晕眩，true为晕眩
		bool xuanyun;
	//BUFF是否禁锢，true为禁锢
		bool jingu;
	//BUFF对速度的改变，原速度为1，即乘的倍数
		float changeSpeed;
	//每秒对生命值的影响
		int32 changeHP;
	//增加/减少力量（攻击力）
		int32 changePower;
